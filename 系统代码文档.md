## 1. 系统架构概述

### 1.1 技术栈
- **前端框架**：原生JavaScript (Vanilla JS)
- **界面技术**：HTML5 + CSS3
- **数据存储**：内存对象 (模拟数据库)
- **实时更新**：setInterval 定时器
- **事件处理**：DOM事件监听

### 1.2 架构模式
采用**MVC (Model-View-Controller)** 架构模式：

```
┌─────────────────────────────────────────────────────────┐
│                    View Layer (视图层)                   │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐  │
│  │   index.html │  │  style.css   │  │  DOM Elements│  │
│  └──────────────┘  └──────────────┘  └──────────────┘  │
└─────────────────────────────────────────────────────────┘
                            ↕
┌─────────────────────────────────────────────────────────┐
│                 Controller Layer (控制层)                │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐  │
│  │ Event List.  │  │  UI Update   │  │  Validation  │  │
│  │   Functions  │  │  Functions   │  │   Functions  │  │
│  └──────────────┘  └──────────────┘  └──────────────┘  │
└─────────────────────────────────────────────────────────┘
                            ↕
┌─────────────────────────────────────────────────────────┐
│                  Model Layer (数据层)                    │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐  │
│  │   appData    │  │  Data Struct.│  │  Business    │  │
│  │   Object     │  │  Definitions │  │    Logic     │  │
│  └──────────────┘  └──────────────┘  └──────────────┘  │
└─────────────────────────────────────────────────────────┘
```

### 1.3 核心设计原则
- **数据驱动**：UI更新基于数据变化
- **模块化**：功能模块独立，低耦合
- **可维护性**：清晰的代码结构和命名规范
- **可扩展性**：易于添加新功能和模块

### 4.5 告警系统函数

#### detectCoolingAlerts()
**功能**：检测液冷系统告警
**参数**：无
**返回值**：无
**检测项目**：
- CDU漏液检测

**告警流程**：
```javascript
function detectCoolingAlerts() {
    const alerts = [];

    // 遍历所有CDU
    ['cdu1', 'cdu2', 'cdu3'].forEach(cduId => {
        const cdu = appData.coolingData[cduId];

        // 检测漏液
        if (cdu.leak === 'warning' && checkAlertThrottle(alertId)) {
            alerts.push({
                type: 'critical',
                source: `CDU-${cduId.slice(-1)}`,
                message: '检测到漏液异常',
                priority: 'high'
            });
        }
        // ... 其他检测
    });

    // 处理告警
    alerts.forEach(alert => {
        createAlertWorkOrder(alert.type, alert.source, alert.message, alert.priority);
        recordAlert(alert);
    });
}
```

#### checkAlertThrottle(alertId)
**功能**：防止告警重复
**参数**：
- `alertId`: 告警标识符
**返回值**：Boolean (是否允许告警)
**逻辑**：
- 检查上次告警时间
- 5分钟内不重复告警
- 更新最后告警时间

```javascript
function checkAlertThrottle(alertId) {
    const now = Date.now();
    const lastTime = appData.lastAlertTime[alertId] || 0;
    const throttleTime = 5 * 60 * 1000; // 5分钟

    if (now - lastTime < throttleTime) {
        return false; // 不允许告警
    }

    appData.lastAlertTime[alertId] = now;
    return true; // 允许告警
}
```

function exportLogs() {
    const logContent = appData.systemLogs.map(log =>
        `${log.time} [${log.type}] ${log.message} ${log.details}`
    ).join('\n');

    const blob = new Blob([logContent], { type: 'text/plain;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `logs_${new Date().toISOString().split('T')[0]}.txt`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
}
```

    return function executedFunction(...args) {

            clearTimeout(timeout);
            func(...args);
        };
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
    };
}
```

#### IndexedDB存储

request.onupgradeneeded = function(event) {
    const db = event.target.result;

function saveToIndexedDB(storeName, data) {
    const request = indexedDB.open('MaintenanceSystem', 1);
    request.onsuccess = function(event) {
        const db = event.target.result;
        const transaction = db.transaction([storeName], 'readwrite');
        const store = transaction.objectStore(storeName);
        data.forEach(item => store.put(item));
    };
}
